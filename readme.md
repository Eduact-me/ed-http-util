# @eduact/ed-http-util

## Installation

`npm install --save @eduact/ed-http-util`

or using yarn

`yarn add @eduact/ed-http-util`

## Usage

```javascript
import Http from "@eduact/ed-http-util"

Http.respond(
    responseObject: any,
    data?: object,
    message: string = '',
    statusCode: number = 200,
    meta?: object,
  )
...
```
